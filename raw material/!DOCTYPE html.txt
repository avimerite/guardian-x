    <script>
        // Update sensor data
        function refreshSensors() {
            fetch('/api/sensors')
                .then(response => response.json())
                .then(data => {
                    document.getElementById('temp').textContent = data.temp.toFixed(1) + ' °C';
                    document.getElementById('hum').textContent = data.hum.toFixed(1) + ' %';
                    document.getElementById('air').textContent = data.air;
                    document.getElementById('dist').textContent = data.dist + ' cm';
                    document.getElementById('last-update').textContent = new Date().toLocaleTimeString();
                    showStatus('Sensor data updated', 'success');
                })
                .catch(error => {
                    showStatus('Failed to get sensor data', 'error');
                });
        }

        // Send command to robot
        function sendCommand(cmd, val) {
            fetch('/api/command', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ cmd: cmd, val: val })
            })
            .then(response => response.json())
            .then(data => {
                showStatus(data.message, data.status);
            })
            .catch(error => {
                showStatus('Command failed', 'error');
            });
        }

        // Set servo position
        function setServo(servoId, angle) {
            sendCommand('servo', `${servoId},${angle}`);
            document.getElementById(`servo${servoId}-value`).textContent = angle + '°';
        }

        // Test webcam
        function testWebcam() {
            fetch('/api/webcam_test')
                .then(response => response.json())
                .then(data => {
                    showStatus(`Webcam: ${data.message}`, data.status === 'working' ? 'success' : 'info');
                    updateWebcamStatus(data.status === 'working');
                })
                .catch(error => {
                    showStatus('Webcam test failed', 'error');
                    updateWebcamStatus(false);
                });
        }

        // Restart camera
        function restartCamera() {
            showStatus('Restarting camera...', 'info');
            fetch('/api/camera/restart')
                .then(response => response.json())
                .then(data => {
                    showStatus(data.message, data.status);
                    // Reload the video feed after a short delay
                    setTimeout(() => {
                        document.getElementById('webcam-img').src = '/video_feed?t=' + new Date().getTime();
                    }, 2000);
                })
                .catch(error => {
                    showStatus('Camera restart failed', 'error');
                });
        }

        // Set camera quality
        function setCameraQuality(quality) {
            fetch('/api/camera/settings', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ setting: 'resolution', value: quality })
            })
            .then(response => response.json())
            .then(data => {
                showStatus(data.message, data.status);
                // Refresh the video feed
                setTimeout(() => {
                    document.getElementById('webcam-img').src = '/video_feed?t=' + new Date().getTime();
                }, 1000);
            })
            .catch(error => {
                showStatus('Quality change failed', 'error');
            });
        }

        // Check system status
        function checkSystemStatus() {
            fetch('/api/status')
                .then(response => response.json())
                .then(data => {
                    const statusMsg = `Arduino: ${data.arduino_connected ? 'Connected' : 'Disconnected'} | Webcam: ${data.webcam_connected ? 'Live' : 'Offline'}`;
                    showStatus(statusMsg, 'info');
                    updateWebcamStatus(data.webcam_connected);
                    
                    // Update system indicator
                    const systemIndicator = document.getElementById('system-indicator');
                    const systemText = document.getElementById('system-text');
                    if (data.arduino_connected || data.webcam_connected) {
                        systemIndicator.className = 'status-indicator connected';
                        systemText.textContent = 'System Ready';
                    } else {
                        systemIndicator.className = 'status-indicator disconnected';
                        systemText.textContent = 'System Offline';
                    }
                })
                .catch(error => {
                    showStatus('Status check failed', 'error');
                });
        }

        // Update webcam status
        function updateWebcamStatus(isWorking) {
            const indicator = document.getElementById('webcam-indicator');
            const text = document.getElementById('webcam-text');
            const status = document.getElementById('webcam-status');
            
            if (isWorking) {
                indicator.className = 'status-indicator webcam-active';
                text.textContent = 'Webcam Live';
                status.textContent = 'LIVE';
                status.style.background = 'rgba(52, 152, 219, 0.9)';
            } else {
                indicator.className = 'status-indicator disconnected';
                text.textContent = 'Webcam Offline';
                status.textContent = 'OFFLINE';
                status.style.background = 'rgba(231, 76, 60, 0.9)';
            }
        }

        // Show status message
        function showStatus(message, type) {
            const statusDiv = document.getElementById('command-status');
            statusDiv.textContent = message;
            statusDiv.className = `command-status status-${type}`;
            
            setTimeout(() => {
                statusDiv.textContent = '';
                statusDiv.className = 'command-status';
            }, 3000);
        }

        // Webcam error handling
        function setupWebcamHandlers() {
            const webcamImg = document.getElementById('webcam-img');
            
            webcamImg.onerror = function() {
                updateWebcamStatus(false);
                showStatus('Webcam feed error - attempting recovery...', 'error');
                // Try to restart camera after error
                setTimeout(restartCamera, 3000);
            };

            webcamImg.onload = function() {
                updateWebcamStatus(true);
            };
        }

        // Auto-refresh sensors every 3 seconds
        setInterval(refreshSensors, 3000);
        
        // Initial load
        document.addEventListener('DOMContentLoaded', function() {
            setupWebcamHandlers();
            refreshSensors();
            checkSystemStatus();
        });
    </script>
