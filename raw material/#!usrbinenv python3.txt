#!/usr/bin/env python3
"""
Simple & Reliable Camera Server
Uses Flask + OpenCV directly - no mjpg-streamer complications!
"""

from flask import Flask, Response
import cv2
import threading
import time
import logging

# Set up logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger(__name__)

app = Flask(__name__)

class Camera:
    def __init__(self):
        self.camera = None
        self.frame = None
        self.is_running = False
        self.thread = None
        
    def start(self):
        """Start camera capture thread"""
        if self.is_running:
            return True
            
        logger.info("🚀 Starting camera...")
        
        # Try different camera indices
        for camera_index in [0, 1, 2]:
            try:
                logger.info(f"🔍 Trying camera index {camera_index}...")
                self.camera = cv2.VideoCapture(camera_index)
                
                if self.camera.isOpened():
                    # Set camera properties
                    self.camera.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
                    self.camera.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
                    self.camera.set(cv2.CAP_PROP_FPS, 20)
                    
                    # Test frame capture
                    ret, test_frame = self.camera.read()
                    if ret and test_frame is not None:
                        logger.info(f"✅ Camera {camera_index} connected! Resolution: {test_frame.shape[1]}x{test_frame.shape[0]}")
                        self.is_running = True
                        self.thread = threading.Thread(target=self._capture_loop, daemon=True)
                        self.thread.start()
                        return True
                    else:
                        self.camera.release()
                        self.camera = None
                        
            except Exception as e:
                logger.error(f"❌ Camera {camera_index} failed: {e}")
                if self.camera:
                    self.camera.release()
                    self.camera = None
        
        logger.error("❌ No camera found!")
        return False
    
    def _capture_loop(self):
        """Continuous frame capture"""
        error_count = 0
        max_errors = 5
        
        while self.is_running:
            try:
                if self.camera and self.camera.isOpened():
                    ret, frame = self.camera.read()
                    if ret:
                        self.frame = frame
                        error_count = 0  # Reset error count on success
                    else:
                        error_count += 1
                        logger.warning(f"⚠️ Frame capture failed ({error_count}/{max_errors})")
                        
                        if error_count >= max_errors:
                            logger.error("🔄 Too many errors, attempting camera restart...")
                            self._restart_camera()
                            error_count = 0
                else:
                    logger.warning("⚠️ Camera not available, attempting restart...")
                    self._restart_camera()
                    
            except Exception as e:
                logger.error(f"❌ Capture error: {e}")
                error_count += 1
                time.sleep(0.1)
            
            time.sleep(0.03)  # ~30 FPS
    
    def _restart_camera(self):
        """Restart camera connection"""
        if self.camera:
            self.camera.release()
            self.camera = None
        time.sleep(1)
        self.start()
    
    def get_frame(self):
        """Get current frame"""
        if self.frame is not None:
            # Add timestamp and status to frame
            frame_with_text = self.frame.copy()
            timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
            cv2.putText(frame_with_text, f"Robot Camera - {timestamp}", (10, 30), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 255), 2)
            cv2.putText(frame_with_text, "LIVE - Simple Camera Server", (10, 460), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
            
            # Encode as JPEG
            ret, jpeg = cv2.imencode('.jpg', frame_with_text, [cv2.IMWRITE_JPEG_QUALITY, 80])
            if ret:
                return jpeg.tobytes()
        return None
    
    def stop(self):
        """Stop camera"""
        logger.info("🛑 Stopping camera...")
        self.is_running = False
        if self.camera:
            self.camera.release()
            self.camera = None
        logger.info("✅ Camera stopped")

# Global camera instance
camera = Camera()

def generate_frames():
    """Generate video frames"""
    while True:
        try:
            frame_bytes = camera.get_frame()
            if frame_bytes:
                yield (b'--frame\r\n'
                       b'Content-Type: image/jpeg\r\n\r\n' + frame_bytes + b'\r\n')
            else:
                # Send placeholder frame when no camera
                yield from generate_placeholder()
        except Exception as e:
            logger.error(f"Stream error: {e}")
            yield from generate_placeholder()
        time.sleep(0.03)

def generate_placeholder():
    """Generate placeholder frame"""
    import numpy as np
    while True:
        # Create a simple placeholder
        frame = np.ones((480, 640, 3), dtype=np.uint8) * 50
        cv2.putText(frame, "🤖 ROBOT CAMERA SERVER", (120, 150), 
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 2)
        cv2.putText(frame, "Camera Starting...", (200, 200), 
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 200, 100), 1)
        cv2.putText(frame, "Please wait or check camera connection", (120, 240), 
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (200, 200, 255), 1)
        
        ret, jpeg = cv2.imencode('.jpg', frame)
        if ret:
            yield (b'--frame\r\n'
                   b'Content-Type: image/jpeg\r\n\r\n' + jpeg.tobytes() + b'\r\n')
        time.sleep(1)

@app.route('/')
def index():
    return {
        "status": "Camera Server Running",
        "camera_connected": camera.is_running,
        "endpoints": {
            "video_feed": "/video_feed",
            "status": "/status"
        }
    }

@app.route('/video_feed')
def video_feed():
    return Response(generate_frames(),
                    mimetype='multipart/x-mixed-replace; boundary=frame')

@app.route('/status')
def status():
    return {
        "camera_connected": camera.is_running,
        "service": "simple_camera_server"
    }

@app.route('/start')
def start_camera():
    success = camera.start()
    return {"success": success, "message": "Camera started" if success else "Camera failed to start"}

@app.route('/stop')
def stop_camera():
    camera.stop()
    return {"success": True, "message": "Camera stopped"}

if __name__ == '__main__':
    logger.info("🚀 SIMPLE CAMERA SERVER STARTING...")
    logger.info("📹 Video Feed: http://localhost:5000/video_feed")
    logger.info("📊 Status: http://localhost:5000/status")
    logger.info("💡 This server is much more reliable than mjpg-streamer!")
    
    # Start camera automatically
    camera.start()
    
    try:
        app.run(host='0.0.0.0', port=5000, debug=False)
    except KeyboardInterrupt:
        logger.info("👋 Shutting down...")
    finally:
        camera.stop()
